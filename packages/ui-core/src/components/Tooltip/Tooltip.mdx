## Manipulate Tooltip open state form outside

<Playground>
    <DemoBlockWrapper>
        <DemoTooltipWrapper>
            <DemoTooltipWrapperWithTrigger>
                {({triggerElement, setTriggerElement}) => (
                    <>
                        <span ref={setTriggerElement}>
                            <Button>Open</Button>
                        </span>
                        <OpenStateWrapper>
                            {({ isOpened, openDrop, closeDrop}) => (
                                <Tooltip
                                    onOpen={openDrop}
                                    onClose={closeDrop}
                                    isOpened={isOpened}
                                    trigger="click"
                                    triggerElement={triggerElement}
                                >
                                    <Paragraph>
                                        Please be careful when filling out the forms.
                                    </Paragraph>
                                    <Button onClick={closeDrop}>Close</Button>
                                </Tooltip>
                            )}
                        </OpenStateWrapper>
                    </>
                )}
            </DemoTooltipWrapperWithTrigger>
        </DemoTooltipWrapper>
    </DemoBlockWrapper>
</Playground>
