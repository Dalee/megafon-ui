export class RadioButtonWrapper extends React.Component {
    constructor(props) {
        super(props);
        this.state = {
            selectedOption: '',
        }
    }
    handleChange = value => {
        this.setState({ selectedOption: value });
    }
    render() {
        return this.props.children({
            onChange: this.handleChange,
            selectedOption: this.state.selectedOption,
        })
    }
}
