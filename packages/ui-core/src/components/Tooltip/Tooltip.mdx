## Full controlled from outside

<Playground>
    <DemoBlockWrapper customStyles={{
        gridTemplateColumns: 'auto',
        gap: 50
    }}>
        <DemoTooltipWrapperWithTrigger>
            {({triggerElement, setTriggerElement}) => (
                <OpenStateWrapper>
                    {({ isOpened, handleOpen, handleClose}) => (
                        <>
                            <span ref={setTriggerElement}>
                                Controlled Tooltip
                            </span>
                            <Tooltip
                                trigger="none"
                                isOpened={isOpened}
                                triggerElement={triggerElement}
                            >
                                <Paragraph>
                                    Please be careful when filling out the forms.
                                </Paragraph>
                                <Button onClick={handleClose}>Close</Button>
                            </Tooltip>
                            <Button onClick={handleOpen}>Open</Button>
                        </>
                    )}
                </OpenStateWrapper>
            )}
        </DemoTooltipWrapperWithTrigger>
    </DemoBlockWrapper>
</Playground>
