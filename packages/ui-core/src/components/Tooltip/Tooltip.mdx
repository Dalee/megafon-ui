## Change trigger

<Playground>
    <DemoBlockWrapper customStyles={{
        gridTemplateColumns: 'auto',
        gap: 50
    }}>
        <DemoTooltipWrapperWithTrigger>
            {({triggerElement, setTriggerElement}) => (
            <DemoSelectTriggerWrapper>
                {({ onSelect, currentValue }) => (
                    <ControlledWrapper isOpen={true}>
                        {({
                            isOpened,
                            handleToggle
                        }) => (
                            <>
                                <div>
                                    <span ref={setTriggerElement}>
                                        Tooltip trigger
                                    </span>
                                    <Tooltip
                                        isOpened={isOpened}
                                        trigger={currentValue}
                                        triggerElement={triggerElement}
                                    >
                                        {currentValue}
                                    </Tooltip>
                                </div>
                                <div style={{ width: 250 }}>
                                    <Select
                                        label="Change trigger"
                                        onSelect={onSelect}
                                        placeholder="Choose trigger"
                                        items={triggerTypes}
                                        currentValue={currentValue}
                                    />
                                </div>
                                <Button
                                    fullWidth
                                    onClick={handleToggle}
                                    disabled={currentValue !== Trigger.CONTROLLED}
                                >
                                    Toggle Tooltip
                                </Button>
                            </>
                        )}
                    </ControlledWrapper>
                )}
            </DemoSelectTriggerWrapper>
        )}
        </DemoTooltipWrapperWithTrigger>
    </DemoBlockWrapper>
</Playground>
