# Search

## Properties

<Props of={Search} />

```javascript
export const DemoSearchWrapper: React.FC<IDemoSearchWrapperProps> = ({ children }) => {
    const [value, setValue] = React.useState('');

    const handleSubmit = (value: string): void => {
        setValue(value);
    };

    return (
        <div>
            <p style={{ marginTop: 0 }}>Value: {value}</p>
            {children({ onSubmit: handleSubmit })}
        </div>
    );
};
```

## Usage with emulating incomingData

<Playground>
      {() => {
        const [items, setItems] = React.useState([]);
        const [chosenValue, setValue] = React.useState('');
        const onChange = React.useCallback((query) => {
                if (!query) {
                    return;
                }
                setItems((oldItems) => [...oldItems, { title: query, value: query }]);
            }, [items, setItems]);
        const onSubmit = React.useCallback((value) => setValue(value), [chosenValue, setValue]);
        return (
            <Search
                value={chosenValue}
                items={items}
                placeholder="Type and wait for results"
                onChange={onChange}
                onSubmit={onSubmit}
            />
        )
      }}
</Playground>
